private async _prewarmCurtainWallShaders(): Promise<void> {
    if (this._shadersPrewarmed) return;
    
    // Build a single real CW segment (not a probe) against the reference slab
    // Use the first edge of slabs[0] — same geometry as production
    const refSlab = this._slabs[0];
    const refEdge = this._getSlabEdges(refSlab)[0];
    
    // Build one real segment, invisible, into a detached scene
    const probeScene = new THREE.Scene();
    this._buildSingleWallSegmentInto(refEdge, probeScene);
    
    // Render the probe scene — warms exact PSO variants
    const rpm = window.renderPipelineManager;
    rpm.render(0); // synchronous — PSOs warm before runBatch()
    
    // Dispose probe geometry immediately
    probeScene.traverse(o => {

            o.geometry.dispose();
        }
    });
    
    this._shadersPrewarmed = true;
}
